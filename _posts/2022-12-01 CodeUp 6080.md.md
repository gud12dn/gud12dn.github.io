#문제: 1부터 n까지, 1부터 m까지 숫자가 적힌 서로 다른 주사위 2개를 던졌을 때,나올 수 있는 모든 경우를 출력해보자.
#for문 사용
n,m=map(int,input().split())#두 가지의 정수를 넣어야 함.
for i in range(1,n+1): #두가지의 적힌 서로 다른 주사위의 경우의 수를 알아야 하기 때문에 for 문을 두가지 사용한다.
    for j in range(1,m+1):
        print(i,j)
