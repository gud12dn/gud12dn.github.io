```python
#6088
#문제: 어떤 규칙에 따라 수를 순서대로 나열한 것을 수열(sequences)이라고 한다.
#while문 이용
a,b,n=map(int,input().split())
i,result=1,a
while(i<n):#1부터 시작이므로 n 사용
    result+=b
    i+=1
print(result)
```
