```python
#6090 수 나열하기3
#문제:어떤 규칙에 따라 수를 순서대로 나열한 것을 수열(sequences)이라고 한다.

#예를 들어
#1 -1 3 -5 11 -21 43 ... 은
#1부터 시작해 이전에 만든 수에 -2를 곱한 다음 1을 더해 다음 수를 만든 수열이다.

#while문 이용
a,b,c,n=map(int,input().split())
i,result=1,a
while(i<n):
    result*=b #-2 곱한 다음
    result+=c#1을 더한다 # simply make a code. result= result*b+c 
    i+=1
print(result)
```

    1 -2 1 8
    -85



```python
#for문 이용 
a,b,c,n=map(int,input().split())

    a=a*b+c
```
