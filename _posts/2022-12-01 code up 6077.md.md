```python
# code up 6077
# 정수(1 ~ 100) 1개를 입력받아 1부터 그 수까지 짝수의 합을 구해보자.
#for 문을 이용한 짝수합 구하기
a=int(input())
b=0 #초깃값을 0
for i in range(0,a+1,2): # 0부터 ~ a까지 탐색 후 i는 2씩 증가
    b+=i
print(b)
```
